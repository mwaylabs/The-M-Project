 11 /**
 12  * @class
 13  *
 14  * The root object for ContainerView.
 15  *
 16  */
 17 M.ContainerView = M.View.extend(
 18 /** @scope M.ContainerView.prototype */ {
 19 
 20     /**
 21      * The type of this object.
 22      *
 23      * @property {String}
 24      */
 25     type: 'M.ContainerView',
 26 
 27     /**
 28      * Renders a simple div container and applies css classes if specified.
 29      */
 30     render: function() {
 31         this.html += '<div id="' + this.id + '"' + this.style() + '></div>';      
 32         return this.html;
 33     },
 34 
 35     /**
 36      * Applies some style-attributes to the container view.
 37      */
 38     style: function() {
 39         var html = '';
 40         if(this.cssClass) {
 41             html += ' class="' + this.cssClass + '"';
 42         }
 43         return html;
 44     }
 45 
 46 });
