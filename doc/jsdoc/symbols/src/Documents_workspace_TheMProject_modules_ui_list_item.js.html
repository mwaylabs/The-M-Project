 11 m_require('ui/button.js');
 12 
 13 /**
 14  * @class
 15  *
 16  * The root object for ListItemViews.
 17  *
 18  */
 19 M.ListItemView = M.View.extend(
 20 /** @scope M.ListItemView.prototype */ {
 21 
 22     type: 'M.ListItemView',
 23 
 24     inEditMode: NO,
 25 
 26     /**
 27      * This property determines whether a list item has one single action that is triggered
 28      * once there is a click anywhere inside the list item or if there are specific actions
 29      * defined for single ui elements within one list item.
 30      *
 31      * property {Boolean}
 32      */
 33     hasSingleAction: YES,
 34 
 35     deleteButton: M.ButtonView.design({
 36         icon: 'delete',
 37         renderToDOM: NO,
 38         target: null,
 39         action: '',
 40         value: ''
 41     }),
 42 
 43     internalTarget: null,
 44 
 45     internalAction: 'setActiveListItem',
 46 
 47     listView: null,
 48 
 49     isDivider: NO,
 50 
 51     render: function() {
 52         this.html = '<li id="' + this.id + '"' + this.style();
 53 
 54         this.html += ' onclick="M.EventDispatcher.onClickEventDidHappen(\'click\', \'' + this.id + '\');"';
 55         this.internalTarget = this.listView;
 56 
 57         if(this.isDivider) {
 58             this.html += ' data-role="list-divider"';
 59         }
 60 
 61         this.html += '>';
 62 
 63         if(this.childViews) {
 64             if(this.inEditMode) {
 65                 this.html += '<a href="#">';
 66                 this.renderChildViews();
 67                 this.html += '</a>';
 68 
 69                 this.html += this.deleteButton.render();
 70             } else {
 71                 this.html += '<a href="#">';
 72                 this.renderChildViews();
 73                 this.html += '</a>';
 74             }
 75         } else if(this.value) {
 76             this.html += this.value;
 77         }
 78 
 79         this.html += '</li>';
 80         
 81         return this.html;
 82     },
 83 
 84     renderUpdate: function() {
 85 
 86     },
 87 
 88     /**
 89      * Triggers render() on all children and returns their render result.
 90      */
 91     renderChildViews: function() {
 92         var childViews = $.trim(this.childViews).split(' ');
 93         for(var i in childViews) {
 94             this.html += this[childViews[i]].render();
 95         }
 96         return this.html;
 97     },
 98 
 99     /**
100      * Applies some style-attributes to the list item.
101      */
102     style: function() {
103         var html = '';
104         if(this.cssClass) {
105             html += ' class="' + this.cssClass + '"';
106         }
107         return html;
108     }
109 
110 });
