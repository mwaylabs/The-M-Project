 13 /**
 14  * @class
 15  *
 16  * The root object for ListViews.
 17  *
 18  */
 19 M.ListView = M.View.extend(
 20 /** @scope M.ListView.prototype */ {
 21 
 22     /**
 23      * The type of this object.
 24      *
 25      * @property {String}
 26      */
 27     type: 'M.ListView',
 28 
 29     /**
 30      * Determines whether to remove all item if the list is updated or not.
 31      *
 32      * @property {Boolean}
 33      */
 34     removeItemsOnUpdate: YES,
 35 
 36     /**
 37      * Determines whether to display the list as a divided list or not.
 38      *
 39      * @property {Boolean}
 40      */
 41     isDividedList: NO,
 42 
 43     /**
 44      * If the list view is a divided list, this property can be used to customize the style
 45      * of the list's dividers.
 46      *
 47      * @property {String}
 48      */
 49     cssClassForDivider: null,
 50 
 51     /**
 52      * Determines whether to display the the number of child items for each list item view.
 53      *
 54      * @property {Boolean}
 55      */
 56     isCountedList: NO,
 57 
 58     /**
 59      * If the list view is a counted list, this property can be used to customize the style
 60      * of the list item's counter.
 61      *
 62      * @property {String}
 63      */
 64     cssClassForCounter: null,
 65 
 66     /**
 67      * This property can be used to customize the style of the list view's split view. For example
 68      * the toggleRemove() of a list view uses the built-in split view functionality.
 69      *
 70      * @property {String}
 71      */
 72     cssClassForSplitView: null,
 73 
 74     /**
 75      * The list view's items, respectively its child views.
 76      *
 77      * @property {Object}
 78      */
 79     items: null,
 80 
 81     /**
 82      * Determines whether the list view is currently in edit mode or not. This is mainly used by the
 83      * built-in toggleRemove() functionality. 
 84      *
 85      * @property {Boolean}
 86      */
 87     inEditMode: NO,
 88 
 89     /**
 90      * This property contains all available options for the edit mode. For example the target and action
 91      * of the automatically rendered delete button can be specified using this property.
 92      *
 93      * @property {Object}
 94      */
 95     editOptions: null,
 96 
 97     /**
 98      * Defines if the ListView is rendered with prefixed numbering for each item.
 99      *
100      * @property {Boolean}
101      */
102     isNumberedList: NO,
103 
104     /**
105      * This property contains the list view's template view, the blueprint for every child view.
106      *
107      * @property {Object}
108      */
109     listItemTemplateView: null,
110 
111     /**
112      * Determines whether to display the list view 'inset' or at full width.
113      *
114      * @property {Boolean}
115      */
116     isInsetList: NO,
117 
118     /**
119      * The list view's search bar.
120      *
121      * @property {Object}
122       */
123     searchBar: null,
124 
125     /**
126      * Determines whether or not to display a search bar at the top of the list view. 
127      *
128      * @property {Boolean}
129      */
130     hasSearchBar: NO,
131 
132     /**
133      * If the hasSearchBar property is set to YES, this property determines whether to use the built-in
134      * simple search filters or not. If set to YES, the list is simply filtered on the fly according
135      * to the entered search string. Only list items matching the entered search string will be visible.
136      *
137      * If a custom search behaviour is needed, this property must be set to NO.
138      *
139      * @property {Boolean}
140      */
141     usesDefaultSearchBehaviour: YES,
142 
143     /**
144      * An object containing target and action to be triggered if the search string changes.
145      *
146      * @property {Object}
147      */
148     onSearchStringDidChange: null,
149 
150     /**
151      * This method renders the empty list view either as an ordered or as an unordered list. It also applies
152      * some styling, if the corresponding properties where set.
153      */
154     render: function() {
155         if(this.hasSearchBar && !this.usesDefaultSearchBehaviour) {
156             this.searchBar.isListViewSearchBar = YES;
157             this.searchBar.listView = this;
158             this.searchBar = M.SearchBarView.design(this.searchBar);
159             this.html += this.searchBar.render();
160         }
161 
162         var listTagName = this.isNumberedList ? 'ol' : 'ul';
163         this.html += '<' + listTagName + ' id="' + this.id + '" data-role="listview"' + this.style() + '></' + listTagName + '>';
164         
165         return this.html;
166     },
167 
168     renderChildViews: function() {
169 
170     },
171 
172     addItem: function(item) {
173         $('#' + this.id).append(item);
174     },
175 
176     removeItem: function(id) {
177 
178     },
179 
180     removeAllItems: function() {
181         $('#' + this.id).empty();
182     },
183 
184     renderUpdate: function() {
185 
186         /* Remove all list items if the removeItemsOnUpdate property is set to YES */
187         if(this.removeItemsOnUpdate) {
188             this.removeAllItems();
189         }
190 
191         /* Save this in variable that for later use within an other scope (e.g. _each()) */
192         var that = this;
193 
194         /* Get the list view's content as an object from the assigned content binding */
195         var content = eval(this.contentBinding);
196 
197         /* Get the list view's template view for each list item */
198         var templateView = this.listItemTemplateView;
199 
200         /* If there is an items property, re-assign this to content, otherwise iterate through content itself */
201         if(this.items) {
202             content = content[this.items];
203         }
204 
205         if(this.isDividedList) {
206             _.each(content, function(items, divider) {
207                 that.renderListItemDivider(divider);
208                 that.renderListItemView(items, templateView);
209             });
210         } else {
211             this.renderListItemView(content, templateView);
212         }
213 
214         /* Finally let the whole list look nice */
215         this.themeUpdate();
216     },
217 
218     renderListItemDivider: function(name) {
219         var obj = M.ListItemView.design({});
220         obj.value = name;
221         obj.isDivider = YES,
222         this.addItem(obj.render());
223         obj.theme();
224     },
225 
226     renderListItemView: function(content, templateView) {
227         /* Save this in variable that for later use within an other scope (e.g. _each()) */
228         var that = this;
229 
230         _.each(content, function(item) {
231 
232             /* Create a new object for the current template view */
233             var obj = templateView.design({});
234 
235             /* If item is a model, assign the model's id to the view's modelId property */
236             if(item.type === 'M.Model') {
237                 obj.modelId = item.id;
238             }
239 
240             /* Get the child views as an array of strings */
241             var childViewsArray = obj.childViews.split(' ');
242 
243             /* If the item is a model, read the values from the 'record' property instead */
244             var record = item.type === 'M.Model' ? item.record : item;
245 
246             /* Iterate through all views defined in the template view */
247             for(var i in childViewsArray) {
248                 /* Create a new object for the current view */
249                 obj[childViewsArray[i]] = obj[childViewsArray[i]].design({});
250 
251                 var regexResult = null;
252                 if(obj[childViewsArray[i]].computedValue) {
253                     /* This regex looks for a variable inside the template view (<%= ... %>) ... */
254                     regexResult = /^<%=\s+([.|_|-|$|�|a-zA-Z]+[0-9]*[.|_|-|$|�|a-zA-Z]*)\s*%>$/.exec(obj[childViewsArray[i]].computedValue.valuePattern);
255                 } else {
256                     regexResult = /^<%=\s+([.|_|-|$|�|a-zA-Z]+[0-9]*[.|_|-|$|�|a-zA-Z]*)\s*%>$/.exec(obj[childViewsArray[i]].valuePattern);
257                 }
258 
259                 /* ... if a match was found, the variable is replaced by the corresponding value inside the record */
260                 if(regexResult) {
261                     switch (obj[childViewsArray[i]].type) {
262                         case 'M.LabelView':
263                         case 'M.ButtonView':
264                         case 'M.ImageView':
265                             obj[childViewsArray[i]].value = record[regexResult[1]];
266                             break;
267                     }
268                 }
269             }
270 
271             /* If edit mode is on, render a delete button */
272             if(that.inEditMode) {
273                 obj.inEditMode = that.inEditMode;
274                 obj.deleteButton = obj.deleteButton.design({
275                     modelId: item.id,
276                     target: that.editOptions.target,
277                     action: that.editOptions.action
278                 });
279             }
280 
281             /* set the list view as 'parent' for the current list item view */
282             obj.listView = that;
283 
284             /* Add the current list view item to the list view ... */
285             that.addItem(obj.render());
286 
287             /* ... once it is in the DOM, make it look nice */
288             for(var i in childViewsArray) {
289                 obj[childViewsArray[i]].theme();
290             }
291         });
292     },
293 
294     /**
295      * Triggers rendering engine, e.g. jQuery mobile, to style the list view.
296      */
297     theme: function() {
298         $('#' + this.id).listview();
299         if(this.searchBar) {
300             this.searchBar.theme();
301         }
302     },
303 
304     /**
305      * Triggers rendering engine, e.g. jQuery mobile, to style the button.
306      */
307     themeUpdate: function() {
308         $('#' + this.id).listview('refresh');
309     },
310 
311     /**
312      * This method activates the edit mode and forces the list view to re-render itself
313      * and to display a remove button for every list view item.
314      *
315      * @param {Object} options The options for the remove button.
316      */
317     toggleRemove: function(options) {
318         if(eval(this.contentBinding)) {
319             this.inEditMode = !this.inEditMode;
320             this.editOptions = options;
321             this.renderUpdate();
322         }
323     },
324 
325     setActiveListItem: function(listItemId) {
326         $('#' + this.id).find('li').each(function() {
327             var listItem = M.ViewManager.getViewById($(this).attr('id'));
328             listItem.removeCssClass('ui-btn-active');
329         });
330         M.ViewManager.getViewById(listItemId).addCssClass('ui-btn-active')
331     },
332 
333     /**
334      * Applies some style-attributes to the list view.
335      */
336     style: function() {
337         var html = '';
338         if(this.isDividedList && this.cssClassForDivider) {
339             html += ' data-dividertheme="' + this.cssClassForDivider + '"';
340         }
341         if(this.isInsetList) {
342             html += ' data-inset="true"';
343         }
344         if(this.isCountedList && this.cssClassForCounter) {
345             html += ' data-counttheme="' + this.cssClassForCounter + '"';
346         }
347         if(this.cssClassForSplitView) {
348             html += ' data-splittheme="' + this.cssClassForSplitView + '"';
349         }
350         if(this.hasSearchBar && this.usesDefaultSearchBehaviour) {
351             html += ' data-filter="true"';
352         }
353         return html;
354     }
355 
356 });
