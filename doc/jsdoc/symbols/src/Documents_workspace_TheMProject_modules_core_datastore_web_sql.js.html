 65      save: function(obj) {
 66         console.log('save() called.');
 67 
 68         this.onSuccess = obj.onSuccess;
 69         this.onError = obj.onError;
 70 
 71         /**
 72          * if not already done, initialize db/table first
 73          */
 74         if(!this.isInitialized) {
 75             this.internalCallback = this.save;
 76             this.init(obj.model);
 77             return;
 78         }
 79 
 80 
 81         if(obj.model.state === M.STATE_NEW) { // perform an INSERT
 82 
 83             var sql = 'INSERT INTO ' + obj.model.name + ' (';
 84             for(var prop in obj.model.record) {
 85                 sql += prop + ', ';
 86             }
 87 
 88             sql = sql.substring(0, sql.lastIndexOf(',')) + ') ';
 89             sql += 'VALUES (';
 90 
 91             for(var prop in obj.model.record) {
 92                 /* if property is string or text write value in quotes */
 93                 var pre_suffix = obj.model.__meta[prop].type === 'String' || obj.model.__meta[prop].type === 'Text' ? '"' : '';
 94                 sql += pre_suffix + obj.model.record[prop] + pre_suffix + ', ';
 95             }
 96             sql = sql.substring(0, sql.lastIndexOf(',')) + '); ';
 97 
 98             console.log(sql);
 99 
100             this.performOp(sql, obj, 'INSERT');
101 
102         } else { // perform an UPDATE with id of model
103 
104             var sql = 'UPDATE ' + obj.model.name + ' SET ';
105 
106             for(var prop in obj.model.record) {
107                 var pre_suffix = obj.model.__meta[prop].type === 'String' || obj.model.__meta[prop].type === 'Text' ? '"' : '';
108                 sql += prop + ' = ' + pre_suffix + obj.model.record[prop] + pre_suffix + ', ';
109             }
110             sql = sql.substring(0, sql.lastIndexOf(','));
111             sql += ' WHERE ' + 'ID = ' + obj.model.record.ID;
112 
113             console.log(sql);
114 
115             this.performOp(sql, obj, 'UPDATE');
116         }
117     },

119     /**
120      * performs operation on websql storage: INSERT, UPDATE or DELETE. is used by them (save() & del())
121      *
122      * @param {String} sql The query.
123      * @param {Object} obj The param object. Contains e.g. callbacks (onError & onSuccess)
124      */
125     performOp: function(sql, obj, opType) {
126         var that = this;
127         this.dbHandler.transaction(function(t) {
128             t.executeSql(sql, null, function() {
129                 if(opType === 'INSERT') {
130                     that.queryDbForId(obj.model);
131                 }
132             }, function() { // error callback for SQLStatementTransaction
133                 M.Logger.log('Incorrect statement: ' + sql, M.ERROR)
134             });
135         },
136         function() { // errorCallback
137             /* bind error callback */
138             if (obj.onError && obj.onError.target && obj.onError.action) {
139                 obj.onError = this.bindToCaller(obj.onError.target, obj.onError.target[obj.onError.action]);
140                 obj.onError();
141             } else if (typeof(obj.onError) !== 'function') {
142                 M.Logger.log('Target and action in onError not defined.', M.INFO);
143             }
144         }, function() { // voidCallback (success)
145             /* bind success callback */
146             if (obj.onSuccess && obj.onSuccess.target && obj.onSuccess.action) {
147                 obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action]);
148                 obj.onSuccess();
149             } else if (typeof(obj.onError) !== 'function') {
150                 M.Logger.log('Target and action in onSuccess not defined.', M.INFO);
151             }
152         });        
153     },

155     /**
156      * Deletes the passed model from the database
157      * @param model
158      */
159     del: function(obj) {
160         console.log('del() called.');
161         if(!this.isInitialized) {
162             this.internalCallback = this.del;
163             this.init(obj, this.bindToCaller(this, this.del));
164             return;
165         }
166 
167         var sql = 'DELETE FROM ' + obj.model.name + ' WHERE ID=' + obj.model.record.ID + ';';
168 
169         console.log(sql);
170 
171         this.performOp(sql, obj, 'DELETE');
172     },

175     /**
176      * @param {Object} obj The param object. Includes: model, query and callback(onSuccess, onError) as parameters.
177      */
178     find: function(obj) {
179         console.log('find() called.');
180         
181         this.onSuccess = obj.onSuccess;
182         this.onError = obj.onError;
183         
184         if(!this.isInitialized) {
185             this.internalCallback = this.find;
186             this.init(obj, this.bindToCaller(this, this.find));
187             return;
188         }
189 
190         var sql = 'SELECT ';
191 
192         if(obj.columns) {
193             /* ID column always needs to be in de result relation */
194             if(!(_.include(obj.columns, 'ID'))) {
195                 obj.columns.push('ID');
196             }
197 
198             if(obj.columns.length > 1) {
199                 sql += obj.columns.join(', ');
200             } else if(obj.columns.length == 1) {
201                 sql += obj.columns[0] + ' ';
202             }
203         } else {
204             sql += '* ';
205         }
206         
207         sql += ' FROM ' + obj.model.name;
208 
209         var stmtParameters = [];
210 
211         /* now process constraint */
212         if(obj.constraint) {
213 
214             var n = obj.constraint.statement.split("?").length - 1;
215 
216             /* if parameters are passed we assign them to stmtParameters, the array that is passed for prepared statement substitution*/
217             if(obj.constraint.parameters) {
218 
219                 if(n === obj.constraint.parameters.length) { /* length of parameters list must match number of ? in statement */
220                     sql += obj.constraint.statement;
221                     stmtParameters = obj.constraint.parameters;
222                 } else {
223                     M.Logger.log('Not enough parameters provided for statement.', M.ERROR);
224                     return NO;
225                 }    
226            /* if no ? are in statement, we handle it as a non-prepared statement
227             * => developer needs to take care of it by himself regarding
228             * sql injection => all statements that are constructed with dynamic
229             * input should be done as prepared statements
230             */
231             } else if(n === 0) {
232                 sql += obj.constraint.statement;
233             }
234         }
235 
236         /* now attach order */
237         if(obj.order) {
238             sql += ' ORDER BY ' + obj.order
239         }
240 
241         /* now attach limt */
242         if(obj.limit) {
243             sql += ' LIMIT ' + obj.limit
244         }
245 
246         console.log(sql);
247 
248         var result = [];
249         var that = this;
250         this.dbHandler.readTransaction(function(t) {
251             t.executeSql(sql, stmtParameters, function (tx, res) {
252                 var len = res.rows.length, i;
253                 for (i = 0; i < len; i++) {
254                     /* create model record from result with state valid */
255                     /* $.extend merges param1 object with param2 object*/
256                     result.push(obj.model.createRecord($.extend(res.rows.item(i), {state: M.STATE_VALID}), this));
257                 }
258             }, function(){M.Logger.log('Incorrect statement: ' + sql, M.ERROR)}) // callbacks: SQLStatementErrorCallback
259         }, function(){ // errorCallback
260             /* bind error callback */
261             if(obj.onError && obj.onError.target && obj.onError.action) {
262                 obj.onError = this.bindToCaller(obj.onError.target, obj.onError.target[obj.onError.action]);
263                 obj.onError();
264             } else if (typeof(obj.onError) !== 'function') {
265                 M.Logger.log('Target and action in onError not defined.', M.ERROR);
266             }
267         }, function() { // voidCallback (success)
268             /* bind success callback */
269             if(obj.onSuccess && obj.onSuccess.target && obj.onSuccess.action) {
270                 /* [result] is a workaround for bindToCaller: bindToCaller uses call() instead of apply() if an array is given.
271                  * result is an array, but we what call is doing with it is wrong in this case. call maps each array element to one method
272                  * parameter of the function called. Our callback only has one parameter so it would just pass the first value of our result set to the
273                  * callback. therefor we now put result into an array (so we have an array inside an array) and result as a whole is now passed as the first
274                  * value to the callback.
275                  *  */
276                 obj.onSuccess = that.bindToCaller(obj.onSuccess.target, obj.onSuccess.target[obj.onSuccess.action], [result]);
277                 obj.onSuccess();
278             }
279         });
280     },

293     /**
294      * creates the table corresponding to the model.
295      */
296     createTable: function(obj, callback) {
297         console.log('createTable() called.');
298         var sql = 'CREATE TABLE IF NOT EXISTS '  + obj.model.name
299                     + ' (ID INTEGER PRIMARY KEY ASC AUTOINCREMENT UNIQUE';
300 
301         for(var r in obj.model.__meta) {
302            sql += ', ' + this.buildDbAttrFromProp(obj.model, r);
303         }
304 
305         sql += ');';
306 
307         console.log(sql);
308         
309         if(this.dbHandler) {
310             var that = this;
311             try {
312                 /* transaction has 3 parameters: the transaction callback, the error callback and the success callback */
313                 this.dbHandler.transaction(function(t) {
314                     t.executeSql(sql);
315                 }, null, that.bindToCaller(that, that.handleDbReturn, [obj, callback]));
316             } catch(e) {
317                 M.Logger.log('Error code: ' + e.code + ' msg: ' + e.message, M.ERROR);
318                 return;
319             }
320         } else {
321             M.Logger.log('dbHandler does not exist.', M.ERROR);
322         }
323     },

339     /**
340      * @param {Object} 
341      * @return {String} The string used for db create to represent this property.
342      */
343     buildDbAttrFromProp: function(model, prop) {
344         var type = this.typeMapping[model.__meta[prop].type].toUpperCase();
345 
346         var isReqStr = model.__meta[prop].isRequired ? ' NOT NULL' : '';
347 
348         return prop + ' ' + type + isReqStr;
349     },

352     /**
353      * Queries the WebSql storage for the maximum id that was provided for a table defined by model.name.
354      * @param {Object} model The table's model
355      */
356     queryDbForId: function(model) {
357         var that = this;
358         this.dbHandler.readTransaction(function(t) {
359             var r = t.executeSql('SELECT seq as ID FROM sqlite_sequence WHERE name="' + model.name + '"', [], function (tx, res) {
360                 that.setDbIdOfModel(model, res.rows.item(0).ID);
361             });
362         });
363     },
364 
365     handleDbReturn: function(obj, callback) {
366         console.log('handleDbReturn() called.');
367         this.isInitialized = YES;
368         //console.log(model);
369         //console.log(callback);
370         this.internalCallback(obj, callback);
371     },
372 
373     setDbIdOfModel: function(model, id) {
374         model.record.ID = id;
375     }
