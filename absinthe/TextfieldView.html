example1: M.TextfieldView.extend({
                grid: 'col-xs-12',
                label: 'Label',
                value: '',
                icon: 'fa-rocket',
                placeholder: 'Rocket'
            }),

 backgroundRightTextfieldExample: M.TextfieldView.extend({
                grid: 'col-xs-12',
                label: 'Label',
                value: '',
                cssClass: 'right',
                icon: 'fa-dot-circle-o',
                placeholder: 'Dot'
            }),

                type: 'clear',
